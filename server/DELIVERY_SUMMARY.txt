  TO:   ImprovedGRNEditManager.editGRN(...)

BEFORE:
  const { default: SimpleGRNEditManager } = 
    await import("...SimpleGRNEditManager.js");
  
  cascadeResult = await SimpleGRNEditManager.editReceivedGRN(
    id,
    { ...req.body, originalItems: originalData.items },
    req.body.createdBy || "System"
  );

AFTER:
  const { default: ImprovedGRNEditManager } = 
    await import("...ImprovedGRNEditManager.js");
  
  cascadeResult = await ImprovedGRNEditManager.editGRN(
    id,
    { items: items, notes: notes || req.body.notes },
    req.body.createdBy || "System"
  );


💾 TRANSACTION FLOW
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

SCENARIO: Edit GRN-1 from 100 → 150 units

  Step 1: Start Transaction 🔄
  Step 2: Build item maps (old vs new)
  Step 3: Calculate delta: 150 - 100 = +50
  Step 4: Update CurrentStock: $inc: +50 ✅
  Step 5: Recalculate availableQty ✅
  Step 6: Update InventoryBatch ✅
  Step 7: Log to StockBefore ✅
  Step 8: Update VendorPayment ✅
  Step 9: Update GRN document ✅
  Step 10: Commit Transaction 🎯

  If ANY error: Entire transaction rolls back ✅
